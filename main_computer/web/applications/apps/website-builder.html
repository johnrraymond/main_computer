    Remote Coolify Python runtime service

    Create a Docker Compose service like this in Coolify so the uploaded folder is served by the Main Computer Python runtime.

    services:
  johnrraymond-site:
    image: 'python:3.12-slim'
    restart: unless-stopped
    working_dir: /app
    command:
      - python
      - /app/sites/johnrraymond/.main-computer/runtime/app.py
    environment:
      SITE_ID: johnrraymond
      SITE_NAME: johnrraymond
      SITE_KIND: static-site
      SITE_LANE: production
      MC_SITE_ID: johnrraymond
      MC_RUNTIME_LANE: production
      CONTENT_ROOT: /app/sites
      BLOG_ENABLED: 'true'
      BLOG_PROVIDER: directus
      BLOG_CONTENT_RUNTIME: directus
      BLOG_COLLECTION: posts
      DIRECTUS_URL: 'https://directus-johnrraymond.greatlibrary.io'
      DIRECTUS_PUBLIC_URL: 'https://directus-johnrraymond.greatlibrary.io'
    volumes:
      - '/srv/main-computer/sites/johnrraymond:/app/sites/johnrraymond:ro'
    expose:
      - '8080'
    healthcheck:
      test:
        - CMD-SHELL
        - 'python -c "import sys, urllib.request; sys.exit(0 if urllib.request.urlopen(''http://127.0.0.1:8080/api/site/status'', timeout=5).status == 200 else 1)"'
      interval: 10s
      timeout: 5s
      retries: 12
      start_period: 20s
